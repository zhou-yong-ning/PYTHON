# 文件类型转换

## docx转doc

```python
from win32com import client as wc  # 导入模块
import os
def save_asdoc(A):
    print(A.split('.')[0] + '.doc')
    word = wc.Dispatch("Word.Application")  # 打开word应用程序
    # 必需绝对路径
    doc = word.Documents.Open("C:\\Users\\zhou\\Desktop\\新建文件夹\\{}".format(A))  # 打开word文件
    # 必需绝对路径
    doc.SaveAs("C:\\Users\\zhou\\Desktop\\新建文件夹\\new\\{}".format(A.split('.')[0] + '.doc'),16) # 另存为后缀为".docx"的文件，其中参数16指doc文件
    doc.Close()  # 关闭原来word文件
    word.Quit()


if __name__ == '__main__':
    file_list = os.listdir("C:\\Users\\zhou\\Desktop\\新建文件夹")
    docx_list = [a for a in file_list if a.endswith('.docx')]
    for i in docx_list:
        save_asdoc(i)
```
